#include <fstream>
#include <iostream>
#include <string>

int main() {
    // Шаблоны дней недели и предметов
    const std::string days[7] = { "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday", "Sunday" };
    const std::string subjects[5] = { "Math", "Physics", "Chemistry", "Biology", "History" };

    // Создание и запись в текстовые файлы для каждого дня недели
    for (const auto& day : days) {
        std::ofstream file(day + ".txt");
        for (const auto& subject : subjects) {
            file << subject << std::endl; // Записываем каждый предмет на новой строке
        }
        file.close(); // Закрываем файл после записи
    }

    // Открываем созданные файлы в текстовом редакторе
    for (const auto& day : days) {
        std::string command = "notepad " + day + ".txt"; // Формируем команду для открытия файла
        system(command.c_str()); // Выполняем команду
    }

    return 0; // Завершаем программу
}
